\part*{ОцОфОеОЋОЪ 1}
\section*{ОАОбОЎОБ Ољ'}
ОаОфОЌОЎОю ОЉОюОћОЋОЏОЎОЌ ОћОфОЏОаОАОЋОф ОаОДОЋОЊОфОЎОф\\
ОЎОћОЎ $x_0 \in \mathbb{R}$, ОаОфОЉОЋОаОЪ ОЉОАОЊОеОћ $\left\{\frac{1}{n} \cos\left(n x_0\right)\right\}_{n=1}^{\infty}$ \\
$\cos\left(n x_0\right)$ ОЌОАОЋОъОћ, ОЉОбОЋОЊ $\frac{1}{n} \xrightarrow{n\to\infty}0$, ОюОЏОЪ ОъОЕОЎОДОЋОю "ОЌОАОЋОъОћ ОЏОцОЋОю ОљОцОАОћ" ОћОАОЊОеОћ ОъОфОЏОаОАОф Ою-$0$\\
ОаОеОљОћ ОЏОбОф ОЏОЎ $\left\{\frac{1}{n} \cos \left(n^2x\right) \right\}^\infty_{n=1}$ ОЕОЋОљОцОф Ою$f \equiv 0$ ОЉОъОЎОЊОћ ОЕОЋОЋОћ \\
ОЎОћОЎ $\varepsilon > 0$ ОЋОаОЉОЌОе $N_{\varepsilon} = \frac{1}{\varepsilon}$, ОюОЏОю $\mathbb{N} \ni n > N_{\varepsilon}$ ОЋОюОЏОю $x \in \mathbb{R}$ ОъОфОДОЎОЎОЮ
\[
\left|\frac{1}{n}\cos\left(n^2 x\right) - \underbrace{f\left(x\right)}_{= 0} \right| \leq \frac{1}{n} \underbrace{\left|\cos\left(n^2x\right)\right|}_{\leq 1}\leq \frac{1}{n} < \varepsilon
\]
ОюОЏОЪ ОћОАОЊОеОћ ОъОфОЏОаОАОф ОЉОъ"ОЕ Ою-$f\left(x\right) = 0$\\
$\blacksquare$
\section*{ОАОбОЎОБ ОЉ'}
ОаОфОЌОЎОю ОЉОюОћОЋОЏОЎОЌ ОћОфОЏОаОАОЋОф ОаОДОЋОЊОфОЎОф\\
ОЎОћОЎ $x_0 \in \left[1,4\right]$, ОаОфОЉОЋОаОЪ ОЉОАОЊОеОћ $\left\{n\ln\left(1+\frac{1}{nx_0}\right)\right\}^\infty_{n=1}$
\begin{gather*}
    n\ln\left(1+\frac{1}{nx_0}\right) = \ln\left(\left(1+\frac{1}{nx_0}\right)^n\right)= \ln\left(\left(1+\frac{1}{nx_0}\right)^\frac{nx_0}{x_0}\right) \\
    =  \ln\left(\left(\left(1+\frac{1}{nx_0}\right)^{nx_0}\right)^{\frac{1}{x_0}}\right) = \frac{1}{x_0} \ln\underbrace{\left(\left(1+\frac{1}{nx_0}\right)^{nx_0}\right)}_{\xrightarrow{n\to\infty}e} \xrightarrow{n\to\infty} \frac{1}{x_0} \ln(e) = \frac{1}{x_0}
\end{gather*}
ОюОЏОЪ ОћОАОЊОеОћ ОъОфОЏОаОАОф \textbf{ОаОДОЋОЊОфОЎОф} Ою-$\frac{1}{x}$ \\ ОаОЋОЏОЎОЌ ОћОфОЏОаОАОЋОф ОЉОъОЎОЊОћ ОЕОЋОЋОћ ОЉОљОъОдОбОЋОф ОћОљОцОЎОЋОЪ ОћОЕОДОЋОю\\
ОаОеОдОћ ОюОћОеОљОЋОф ОЏОЎ
\[
    \lim_{n\to\infty}    \sup_{x\in \left[1,4\right]} \left|f_n\left(x\right) - f\left(x\right)\right| = 0
\]
ОаОњОЊОЎОе ОАОЊОеОф ОцОЋОаОДОдОЎОЋОф $g_n(x) = f_n(x) -f(x)$
ОаОњОќОЋОе ОљОф ОћОцОЋОаОДОдОЎОЋОф $g_n(x)$ (ОфОћОюОЎОџ ОћОњОќОЎОеОћ ОќОћОћ ОюОЏОю $n$)
\begin{gather*}
g_n'(x)=    \left(f_n(x) - f(x)\right)' \underset{\text{ОюОЎОаОљОеОЎОЋОф ОћОаОњОќОеОф}}{=} f_n'\left(x\right) - f'\left(x\right)\\
    = \left(n\ln\left(1+\frac{1}{nx}\right)\right)' -\left(\frac{1}{x}\right)' = n \cdot \frac{1}{1+\frac{1}{nx}} \cdot \left(-\frac{1}{nx^2}\right) + \frac{1}{x^2} \\
    = n \cdot \frac{\cancel{nx}}{1+nx} \cdot \left(-\frac{1}{\cancel{nx^2}}\right) + \frac{1}{x^2} =  -\frac{n}{x\left(1+nx\right)} + \frac{1}{x^2} \\
    = \frac{-nx+1+nx}{x^2\left(1+nx\right)} = \frac{1}{x^2\left(1+nx\right)} 
\end{gather*}
ОюОЏОю $n > 0$ ОЋОюОЏОю $x \in \left[1,4\right]$ $x^2\left(1+nx\right)>0$, $\frac{1}{x^2\left(1+nx\right)}>0$, ОЏОюОЋОъОе ОћОаОњОќОеОф ОбОЋОюОћ ОъОъОЕ\\
ОаОфОЉОЋОаОЪ ОЉОћОфОаОћОњОЋОф ОЕОю $g_n(x)$ ОЉОДОдОЋОЋОф\\
ОЏОљОЕОе $x=1$
\begin{gather*}
    \lim_{n\to\infty} \left(f_n\left(1\right) - f\left(1\right)\right) = \lim_{n\to\infty} n\ln\left(1+\frac{1}{n}\right) -\frac{1}{1} = \lim_{n\to\infty} \ln\left(\left(1+\frac{1}{n}\right)^n\right) - 1 \\
    =  \ln(e) - 1 =  1 -1 = 0
\end{gather*}
ОЏОљОЕОе $x=4$
\begin{gather*}
    \lim_{n\to\infty} \left(f_n\left(4\right) - f\left(4\right)\right) = \lim_{n\to\infty} n\ln\left(1+\frac{1}{4n}\right) -\frac{1}{4} \\
    \underset{4\text{ОЏОцОю ОЋОЌОюОЋОДОћ ОЉ-}}{=} \lim_{n\to\infty} \frac{1}{4} \cdot 4n \ln\left(1+\frac{1}{4n}\right) - \frac{1}{4} \\
    =\lim_{n\to\infty} \frac{1}{4} \ln\left(\left(1+\frac{1}{4n}\right)^{4n}\right) - \frac{1}{4} =  \frac{1}{4}\cdot\ln(e) -\frac{1}{4} = \frac{1}{4}-\frac{1}{4} = 0
\end{gather*}
ОъОЏОЎОЋОЋОЪ ОЕ-$g_n(x)$ ОъОЋОаОЋОўОЋОаОЎОф ОбОЋОюОћ ОъОъОЕ ОЉ-$[1,4]$, ОъОфОДОЎОЎОЮ
\[
g_n(1) \leq g_n(x) \leq g_n(4) 
\]
ОюОЏОЪ, ОбОЉОЋОе ОћОбОеОџ ОћОъОЋОЌОюОў ОъОфОДОЎОЎОЮ
\[
\forall x \in \left[1,4\right]: \left|g_n(x)\right| \leq \max\left(\left|g_n(1)\right|,\left|g_n(4)\right|\right)
\]
ОъОЏОЎОЋОЋОЪ ОЕОќОћ ОаОЏОЋОЪ ОюОЏОю $x$, ОЉОцОеОў ОќОћ ОаОЏОЋОЪ ОбОЉОЋОе ОћОАОЋОцОеОЎОъОЋОЮ
\[
0 \leq \sup_{x\in[1,4]}  \left|g_n(x)\right| \leq \max\left(\left|g_n(1)\right|,\left|g_n(4)\right|\right) \xrightarrow{n\to\infty} 0
\]
ОюОЏОЪ ОъОЏОюОю ОћОАОаОЊОЋОЋОЎОЦ'
\[
    \lim_{n\to\infty}    \sup_{x\in[1,4]} \left|g_n(x)\right| = 0
\]
ОЏОюОЋОъОе
\[
    \lim_{n\to\infty}    \sup_{x\in[1,4]} \left|f_n(x) - f(x)\right| = 0
\]
ОъОЏОљОЪ, ОъОћОљОцОЎОЋОЪ ОћОЕОДОЋОю ОюОћОфОЏОаОАОЋОф ОЉОъ"ОЕ. $f_n(x)$ ОъОфОЏОаОАОф ОЉОъ"ОЕ ОюОцОЋОаОДОдОЎОћ $\frac{1}{x}$\\
$\blacksquare$
